12 // Lightfield Decoder
13 // ==================
14 //
15 // This is an example of a simple lightfield decoder. It builds upon the
16 // simple_decoder.c example. It takes an input file containing the compressed
17 // data (in ivf format), treating it as a lightfield instead of a video.
18 // After running the lightfield encoder, run lightfield decoder to decode a
19 // batch of tiles:
20 // examples/lightfield_decoder vase10x10.ivf vase_reference.yuv 4

75 int main(int argc, char **argv) {
76  FILE *outfile = NULL;
77  aom_codec_ctx_t codec;
78  AvxVideoReader *reader = NULL;
79  const AvxInterface *decoder = NULL;
80  const AvxVideoInfo *info = NULL;
81  int num_references;
82  int width, height;
83  aom_image_t reference_images[MAX_EXTERNAL_REFERENCES];
84  size_t frame_size = 0;
85  const unsigned char *frame = NULL;
86  int n, i, j;
87  exec_name = argv[0];
88 
89  if (argc != 4) die("Invalid number of arguments.");
90 
91  reader = aom_video_reader_open(argv[1]);
92  if (!reader) die("Failed to open %s for reading.", argv[1]);
93 
94  if (!(outfile = fopen(argv[2], "wb")))
95  die("Failed to open %s for writing.", argv[2]);
96 
97  num_references = (int)strtol(argv[3], NULL, 0);
98 
99  info = aom_video_reader_get_info(reader);
100  width = info->frame_width;
101  height = info->frame_height;
102 
103  decoder = get_aom_decoder_by_fourcc(info->codec_fourcc);
104  if (!decoder) die("Unknown input codec.");
105  printf("Using %s\n", aom_codec_iface_name(decoder->codec_interface()));
106 
107  if (aom_codec_dec_init(&codec, decoder->codec_interface(), NULL, 0))
108  die_codec(&codec, "Failed to initialize decoder.");
109 
110  // Decode anchor frames.
111  aom_codec_control_(&codec, AV1_SET_TILE_MODE, 0);
112  for (i = 0; i < num_references; ++i) {
113  aom_video_reader_read_frame(reader);
114  frame = aom_video_reader_get_frame(reader, &frame_size);
115  if (aom_codec_decode(&codec, frame, frame_size, NULL))
116  die_codec(&codec, "Failed to decode frame.");
117 
118  if (i == 0) {
119  aom_img_fmt_t ref_fmt = 0;
120  if (aom_codec_control(&codec, AV1D_GET_IMG_FORMAT, &ref_fmt))
121  die_codec(&codec, "Failed to get the image format");
122 
123  // Allocate memory to store decoded references. Allocate memory with the
124  // border so that it can be used as a reference.
125  for (j = 0; j < num_references; j++) {
126  unsigned int border = AOM_BORDER_IN_PIXELS;
127  if (!aom_img_alloc_with_border(&reference_images[j], ref_fmt, width,
128  height, 32, 8, border)) {
129  die("Failed to allocate references.");
130  }
131  }
132  }
133 
134  if (aom_codec_control(&codec, AV1_COPY_NEW_FRAME_IMAGE,
135  &reference_images[i]))
136  die_codec(&codec, "Failed to copy decoded reference frame");
137 
138  aom_codec_iter_t iter = NULL;
139  aom_image_t *img = NULL;
140  while ((img = aom_codec_get_frame(&codec, &iter)) != NULL) {
141  char name[1024];
142  snprintf(name, sizeof(name), "ref_%d.yuv", i);
143  printf("writing ref image to %s, %d, %d\n", name, img->d_w, img->d_h);
144  FILE *ref_file = fopen(name, "wb");
145  aom_img_write(img, ref_file);
146  fclose(ref_file);
147  }
148  }
149 
150  FILE *infile = aom_video_reader_get_file(reader);
151  // Record the offset of the first camera image.
152  const FileOffset camera_frame_pos = ftello(infile);
153 
154  // Process 1 tile.
155  for (n = 0; n < num_tile_lists; n++) {
156  for (i = 0; i <= tile_count_minus_1; i++) {
157  int image_idx = tile_list[n][i].image_idx;
158  int ref_idx = tile_list[n][i].reference_idx;
159  int tc = tile_list[n][i].tile_col;
160  int tr = tile_list[n][i].tile_row;
161  int frame_cnt = -1;
162 
163  // Seek to the first camera image.
164  fseeko(infile, camera_frame_pos, SEEK_SET);
165 
166  // Read out the camera image
167  while (frame_cnt != image_idx) {
168  aom_video_reader_read_frame(reader);
169  frame_cnt++;
170  }
171 
172  frame = aom_video_reader_get_frame(reader, &frame_size);
173 
174  aom_codec_control_(&codec, AV1_SET_TILE_MODE, 1);
175  aom_codec_control_(&codec, AV1D_EXT_TILE_DEBUG, 1);
176  aom_codec_control_(&codec, AV1_SET_DECODE_TILE_ROW, tr);
177  aom_codec_control_(&codec, AV1_SET_DECODE_TILE_COL, tc);
178 
179  av1_ref_frame_t ref;
180  ref.idx = 0;
181  ref.use_external_ref = 1;
182  ref.img = reference_images[ref_idx];
183  if (aom_codec_control(&codec, AV1_SET_REFERENCE, &ref)) {
184  die_codec(&codec, "Failed to set reference frame.");
185  }
186 
187  aom_codec_err_t aom_status =
188  aom_codec_decode(&codec, frame, frame_size, NULL);
189  if (aom_status) die_codec(&codec, "Failed to decode tile.");
190 
191  aom_codec_iter_t iter = NULL;
192  aom_image_t *img = aom_codec_get_frame(&codec, &iter);
193  aom_img_write(img, outfile);
194  }
195  }
196 
197  for (i = 0; i < num_references; i++) aom_img_free(&reference_images[i]);
198  if (aom_codec_destroy(&codec)) die_codec(&codec, "Failed to destroy codec");
199  aom_video_reader_close(reader);
200  fclose(outfile);
201 
202  return EXIT_SUCCESS;
203 }
